a.2)
#include <stdio.h>
int main()
{
printf(“El numero es %d \n”, 15);
printf(“El numero es %i \n”, 15);
printf(“El numero es %x \n”, 15);
printf(“El numero es %o \n”, 15);
return 0;
}

Esta aplicación imprime el número 15 en distintos sistemas, el primero es un número decimal, el segundo es un número decimal también, el tercer número es un hexadecimal y el último es un número octal.
